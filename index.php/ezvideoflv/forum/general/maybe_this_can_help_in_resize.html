Hello community.

Thank you very much Damien, for this great ezpublish extension.

I did some changes that allow the extension to resize the uploaded files:

Example: width:700 / height:500

Add to ezvideoflv.ini.append.php in [Convert] section:



# Size


Options[]=-s <width>x<height>


In [FLVPlayer] section:



Width=700


Height=500


The function buildCommandLine in file extension/ezvideoflv/datatypes/ezvideoflv/ezvideoflv.php look as follow:




   /**


    * Build the command line according to settings


    * @param $original_file : original full file path


    * @param $converted_file : converted full file path


    */


   static function buildCommandLine( $original_file, $converted_file )


   {


       $ini = eZINI::instance( 'ezvideoflv.ini' );


       $program = $ini->variable( 'Converter', 'Program' );


       $options = $ini->variable( 'Converter', 'Options' );


       $width = $ini->variable( 'FLVPlayer' , 'Width' );


       $height = $ini->variable( 'FLVPlayer' , 'Height' );


       $command = $program;


       $array_search = array();


       $array_search[] = '<original_file>';


       $array_search[] = '<converted_file>';


       $array_search[] = '<width>';


       $array_search[] = '<height>';


       $array_replace = array();


       $array_replace[] = $original_file;


       $array_replace[] = $converted_file;


       $array_replace[] = $width;


       $array_replace[] = $height;


       foreach( $options as $opt )


       {


           $optStr = str_replace( $array_search, $array_replace, $opt );


           $command .= ' ' . $optStr;


       }


       return $command;


   }


Still have to view the video not greater than our values

Template file extension/ezvideoflv/design/standard/templates/content/datatype/view/ezvideoflv.tpl look like this:




{**


 * $Id: ezvideoflv.tpl 24 2009-10-04 15:22:02Z dpobel $


 * $HeadURL: http://svn.projects.ez.no/ezvideo...content/datatype/view/ezvideoflv.tpl $


 *}


{def $width=$attribute.content.width


    $height=$attribute.content.height


    $flv=concat( 'video/flv/', $attribute.contentobject_id,"/",$attribute.content.contentobject_attribute_id )


    $preview=concat( 'video/preview/', $attribute.contentobject_id,"/",$attribute.content.contentobject_attribute_id )


    $player=ezini( 'FLVPlayer', 'File', 'ezvideoflv.ini' )


    $options=ezini( 'FLVPlayer', 'Options', 'ezvideoflv.ini' )


    $iniwidth=ezini( 'FLVPlayer', 'Width', 'ezvideoflv.ini' )


    $iniheight=ezini( 'FLVPlayer', 'Height', 'ezvideoflv.ini' )


    $opt_string=''}


 

 

{if or( gt($width, $iniwidth ), eq( $width, 0 ) )}


   {undef $width}


   {def $width=$iniwidth}


{/if}


 

{if or( gt($height, $iniheight), eq( $height, 0 ) )}


   {undef $height}


   {def $height=$iniheight}


{/if}


 

{if $attribute.has_content}


   {if $attribute.content.has_flv}


       {foreach $options as $key => $value}


           {set opt_string=concat( $opt_string, '&amp;', $key, '=', $value )}


       {/foreach}


       <object type="application/x-shockwave-flash" data={$player|ezdesign()} width="{$width}" height="{$height}">


           <param name="movie" value={$player|ezdesign()} />


           <param name="allowFullScreen" value="true" />


           <param name="FlashVars" value="flv={$flv|ezurl( 'no' )}&amp;startimage={$preview|ezurl( 'no' )}{$opt_string}" />


       </object>


   {else}


   <p>{'FLV version not yet generated'|i18n( 'ezvideoflv/datatype' )}


   {/if}


   <p><a href={concat( "content/download/", $attribute.contentobject_id, "/", $attribute.content.contentobject_attribute_id, "/", $attribute.content.original_filename )|ezurl}>{'Download the original video file'|i18n( 'ezvideoflv/datatype' )}</a></p>


{else}


   <p>{'No file'|i18n( 'ezvideoflv/datatype' )}</p>


{/if}


{undef $width $height $iniwidth $iniheight $flv $opt_string $options $player $preview}
