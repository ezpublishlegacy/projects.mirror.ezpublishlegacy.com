 $videos_array = array();

 foreach ($vals as $elem) {    

 

   if ($elem['tag'] == 'ENTRY' && $elem['type'] == 'open') {


     $video_array = array();

     if ($video_array != null) {


       $videos_array[$video_array['id']] = $video_array;

   } else if ($elem['tag'] == 'MEDIA:THUMBNAIL') {   

    foreach ($elem['attributes'] as $attr => $d) { 

       if( $attr == 'URL') {  

       if( $elem['attributes']['HEIGHT'] == '240') { 

       $video_array['thumbnail_url'] = $d; } } }


   } else if ($elem['tag'] == 'MEDIA:PLAYER') {


       foreach ($elem['attributes'] as $attr) {


       $video_array['url'] = $attr; }


   } else if ($elem['tag'] == 'MEDIA:DESCRIPTION') {


     $video_array['description'] = $elem['value'];


   } else if ($elem['tag'] == 'TITLE') {


     $video_array['title'] = $elem['value'];


   } else if ($elem['tag'] == 'MEDIA:KEYWORDS') {


     $video_array['tags'] = $elem['value'];


   } else if ($elem['tag'] == 'PUBLISHED') {


     $video_array['upload_time'] = $elem['value'];


   } else if ($elem['tag'] == 'YT:STATISTICS') { 

    foreach ($elem['attributes'] as $attr => $d) {


       if( $attr == 'VIEWCOUNT') {


       $video_array['view_count'] = $d;  } }


     } else if ($elem['tag'] == 'GD:RATING') {
